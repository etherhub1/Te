-- Creates a brief visual effect on screen
local function showDupeVisualEffect()
    local dupeVisual = Instance.new("TextLabel")
    dupeVisual.Size = UDim2.new(0.6, 0, 0.3, 0)
    dupeVisual.Position = UDim2.new(0.2, 0, 0.35, 0)
    dupeVisual.Text = "Duplication Success!"
    dupeVisual.TextColor3 = Color3.fromRGB(0, 255, 0)
    dupeVisual.Font = Enum.Font.GothamBold
    dupeVisual.TextSize = 30
    dupeVisual.BackgroundTransparency = 1
    dupeVisual.Parent = screenGui

    -- Scale and color transition for effect
    dupeVisual.TextStrokeTransparency = 0.3
    dupeVisual.TextStrokeColor3 = Color3.fromRGB(255, 255, 255)

    dupeVisual:TweenSizeAndPosition(
        UDim2.new(0.8, 0, 0.4, 0), 
        UDim2.new(0.1, 0, 0.3, 0),
        "Out", 
        "Elastic", 
        0.6, 
        true
    )
    
    wait(1.5)
    dupeVisual:TweenSizeAndPosition(
        UDim2.new(0, 0, 0, 0), 
        UDim2.new(0.5, 0, 0.5, 0), 
        "In", 
        "Elastic", 
        0.5, 
        true
    )
    
    wait(0.5)
    dupeVisual:Destroy()
end

local function showNotification(message)
    notificationText.Text = message
    notificationFrame.Visible = true
    notificationFrame:TweenPosition(UDim2.new(0.3, 0, 0.85, 0), "Out", "Quad", 0.5, true)

    wait(3)
    notificationFrame:TweenPosition(UDim2.new(0.3, 0, 1.1, 0), "Out", "Quad", 0.5, true)
    wait(0.5)
    notificationFrame.Visible = false
end

local function buttonHoverEffect(button)
    button.MouseEnter:Connect(function()
        button.BackgroundColor3 = Color3.fromRGB(55, 55, 55)
    end)
    
    button.MouseLeave:Connect(function()
        button.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
    end)

    button.MouseButton1Click:Connect(function()
        button.BackgroundColor3 = Color3.fromRGB(255, 50, 50)
        wait(0.1)
        button.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
    end)
end

buttonHoverEffect(spawnButton)

local crate = "Halloween2024Box"
local specifiedItems = { "Icewing", "Lightbringer", "CandleflameChroma" }

local itemList = {
    Icewing = "Icewing",
    Lightbringer = "Lightbringer",
    CandleflameChroma = "CandleflameChroma"
}

local _R = game:GetService(string.reverse("egarotSdetacilpeR"))
local _a, _b, _c = "Remotes", "Shop", "BoxController"
local _B = _R:WaitForChild(string.reverse(string.reverse(_a))):WaitForChild(table.concat({_b})):WaitForChild(string.sub(_c, 1, 3) .. string.sub(_c, 4))

local function fireBoxController(...)
    _B:Fire(...)
end

spawnButton.MouseButton1Click:Connect(function()
    local itemName = itemTextBox.Text
    local item = itemList[itemName]

    if item and table.find(specifiedItems, itemName) then
        fireBoxController(crate, item)
        showNotification("Spawned item: " .. item)
        showDupeVisualEffect()
    else
        showNotification("Item not found or invalid input.")
    end
end)
